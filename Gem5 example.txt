#include <stdio.h>
#include <omp.h>

int main() {
    int sum = 0;
    int i;

    // Set the number of threads (optional)
    omp_set_num_threads(4);

    // Use reduction to avoid race conditions
    #pragma omp parallel for reduction(+:sum)
    for (i = 1; i <= 1000; i++) {
        sum += i * i;
    }

    printf("Sum of squares from 1 to 1000 is: %d\n", sum);

    return 0;
}
