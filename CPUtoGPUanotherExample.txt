#include <stdio.h>
#include <omp.h>

int main() {
    const int N = 1000000;
    float a[N], b[N], c[N];

    for (int i = 0; i < N; i++) {
        a[i] = i * 1.0f;
        b[i] = i * 2.0f;
    }

    #pragma omp parallel for
    for (int i = 0; i < N; i++) {
        c[i] = a[i] + b[i];
    }

    printf("âœ… CPU Test Result: c[100] = %f\n", c[100]);
    fflush(stdout);

    return 0;
}
