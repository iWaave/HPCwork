#include <stdio.h>
#include <omp.h>

int main() {
    // Set number of threads (optional, you can also set it via OMP_NUM_THREADS)
    omp_set_num_threads(4);

    // Parallel region starts here
    #pragma omp parallel
    {
        int thread_id = omp_get_thread_num();
        printf("Hello, World! from thread %d\n", thread_id);
    }

    return 0;
}
