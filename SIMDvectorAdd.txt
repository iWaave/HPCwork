#include <iostream>
#include <vector>
#include <chrono> // Timing library

using namespace std;

int main() {
    const int size = 1000000;
    vector<float> A(size, 1.5f);
    vector<float> B(size, 2.5f);
    vector<float> C(size, 0.0f);

    auto start = chrono::high_resolution_clock::now(); // Start time

    #pragma omp simd
    for (int i = 0; i < size; ++i) {
        C[i] = A[i] + B[i];
    }

    auto end = chrono::high_resolution_clock::now(); // End time
    chrono::duration<double> duration = end - start;

    // Output just to verify correctness of result (showing only first 5)
    cout << "Sample results: ";
    for (int i = 0; i < 5; ++i)
        cout << C[i] << " ";
    cout << "\nTime taken with SIMD: " << duration.count() << " seconds\n";

    return 0;
}
